        using namespace std;

        int main() {
            double a, b, c, discrimin;

            printf("Mettere cofficiente A, B e C nell ordine seguente: \n");
            scanf("%lf %lf %lf", &a, &b, &c);

            if (a == 0){
                printf("La soluzione dell equazione: %lf\n", (-c / b));
            } else {
                discrimin = b * b - (4 * a * c);

                if (discrimin < 0){
                    printf("Questa equazione non ha soluzioni.\n");
                } else {
                    if (discrimin == 0){
                        printf("L unico punto di questa equazione e (%lf ; 0)\n", -b / (2 * a));
                    } else {
                        double root1 = (-b + sqrt(discrimin)) / (2 * a);
                        double root2 = (-b - sqrt(discrimin)) / (2 * a);
                        printf("Le due soluzioni dell equazione sono: %lf e %lf.\n", root1, root2);
                    }
                }
            }
            return 0;
        }
